<script>
    let ingredientes = [];

    function agregarIngrediente() {
        // Obtener valores
        let nombre = document.getElementById('nombre').value;
        let pCompra = parseFloat(document.getElementById('precio_compra').value);
        let qCompra = parseFloat(document.getElementById('cantidad_compra').value);
        let qUsada = parseFloat(document.getElementById('cantidad_usada').value);

        if(!nombre || !pCompra || !qCompra || !qUsada) {
            alert("Por favor completa todos los datos del ingrediente");
            return;
        }

        // Regla de tres simple para el costo
        let costoReal = (pCompra / qCompra) * qUsada;

        ingredientes.push({ nombre, costo: costoReal });
        
        // Limpiar inputs
        document.getElementById('nombre').value = '';
        document.getElementById('cantidad_usada').value = '';
        document.getElementById('nombre').focus();

        renderizar();
    }

    function borrar(index) {
        ingredientes.splice(index, 1);
        renderizar();
    }

    function renderizar() {
        let lista = document.getElementById('lista-ingredientes');
        lista.innerHTML = '';
        let sumaIngredientes = 0;

        ingredientes.forEach((ing, index) => {
            sumaIngredientes += ing.costo;
            lista.innerHTML += `
                <div>
                    <span>${ing.nombre}</span>
                    <b>$${ing.costo.toFixed(2)}</b>
                    <button class="delete" onclick="borrar(${index})">X</button>
                </div>
            `;
        });

        calcularTotal(sumaIngredientes);
    }

    function calcularTotal(subtotalIngredientes = null) {
        // Si no pasamos el subtotal, lo recalculamos del array
        if (subtotalIngredientes === null) {
            subtotalIngredientes = ingredientes.reduce((acc, el) => acc + el.costo, 0);
        }

        let extras = parseFloat(document.getElementById('gastos_fijos').value) || 0;
        let margen = parseFloat(document.getElementById('margen').value) || 0;

        let costoTotal = subtotalIngredientes + extras;
        // F√≥rmula de venta: Costo + (Costo * Porcentaje)
        let precioVenta = costoTotal * (1 + (margen / 100));

        document.getElementById('costo-total').innerText = '$' + costoTotal.toFixed(2);
        document.getElementById('precio-venta').innerText = '$' + precioVenta.toFixed(2);
    }
</script>
